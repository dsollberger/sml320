8.16

a

library("rstan")
bb_model <- "
  data {
    int<lower = 0, upper = 1000> Y;
  }
  parameters {
    real<lower = 0, upper = 1> pi;
  }
  model {
    Y ~ binomial(1000, pi);
    pi ~ beta(1, 2);
  }
"

climate_change_simulation <- stan(model_code = bb_model,
                                  data = list(Y = 150), 
                                  chains = 4, 
                                  iter = 5000*2, 
                                  refresh = 0,
                                  seed = 320)

b
